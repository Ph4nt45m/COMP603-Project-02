CREATE TABLE BOOKINGLIST(
BOOKINGDATE VARCHAR(15),
BOOKINGLEAVE VARCHAR(15),
FIRSTNAME VARCHAR(50),
SURNAME VARCHAR(50),
ROOMTYPE VARCHAR(15),
STUDENTDISCOUNT BOOLEAN,
CHILDDISCOUNT BOOLEAN,
COST DOUBLE,
PHONENUMBER VARCHAR(15),
EMAIL VARCHAR(100)
);

CREATE TABLE FAQ(
QUSTION VARCHAR(200),
ANSWER VARCHAR(250)
);

CREATE TABLE FLAGGEDBOOKINGS(
BOOKINGDATE DATE,
FIRSTNAME VARCHAR(50),
SURNAME VARCHAR(50),
ROOMTYPE VARCHAR(15),
STUDENTDISCOUNT BOOLEAN,
CHILDDISCOUNT BOOLEAN,
FAMILYDISCOUNT BOOLEAN,
MOBILENUMBER VARCHAR(13) DEFAULT '+64-',
EMAIL VARCHAR(100)
);

CREATE TABLE OTHERQUESTIONS(
ASKCOUNT INT,
QUESTION VARCHAR(200),
ANSWER VARCHAR(250)
);
